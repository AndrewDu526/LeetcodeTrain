## 3. Code Implementation

    class Solution {
    
        public int maxDepth(TreeNode root) {
            return dfs(root);
        }
    
        private int dfs(TreeNode node){
            if(node != null){return Math.max(dfs(node.left), dfs(node.right))+1;}
            return 0;
        }
    }
